// Build AI_TEXT_EDIT intent data with optional HTML field (EXTR-02)
function buildEditIntentData() {
  var data = { text: lastSelectedText };
  if (lastSelectedHtml && lastSelectedHtml.length > 0) {
    data.html = lastSelectedHtml;
    data.format = "html";

}
