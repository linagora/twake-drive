function findActionConfig(actionId) {
  if (actionId === 'free-prompt') return FREE_PROMPT_CONFIG
  for (const action of SCRIBE_ACTIONS) {
    if (action.id === actionId) return action
    if (action.children) {
      for (const child of action.children) {
        if (child.id === actionId) return child
      }
    }
  }
  if (actionId.startsWith('translate-')) {
    const translateChildren = buildTranslateChildren('')
    for (const child of translateChildren) {
      if (child.id === actionId) return child
    }
  }
  return null
}

**2. `buildMessages(actionId, selectedText, label, extra)` function:**
- Returns an array of `[{role: 'system', content: SYSTEM_PROMPT}, {role: 'user', content: ...}]`
- For `actionId === 'free-prompt'`: user content is the guardrail template:
  `"Apply the following instruction to the text below. Return only the modified text.\n\nInstruction: ${label}\n\nText: ${selectedText}"`
  (Per locked decision in CONTEXT.md)
- For all other actions:
  - Reuse `findActionConfig` logic from mockTransform.js (copy the function — it searches SCRIBE_ACTIONS, children, and dynamic translate children)
  - Get `action.prompt`, replace `{selectedText}` with the actual selected text
  - If `actionId === 'translate-custom'`, replace `{language}` with `label` (which contains the user-typed language name — per Pitfall 5 in research)
  - If `extra?.language` exists, also replace `{language}` with `extra.language`
- Return `[systemMsg, {role: 'user', content: interpolatedPrompt}]`

**3. `callScribeAI(client, messages, { signal } = {})` async function:**
- Calls `client.stackClient.fetchJSON('POST', '/ai/v1/chat/completions', { messages }, { signal })`
  (Per research: use fetchJSON directly instead of chatCompletion() — chatCompletion() swallows AbortError and doesn't pass signal to fetch options)
- Extract content: `response?.content || response?.choices?.[0]?.message?.content`
  (Defensive double-check pattern from AIAssistantPanel)
- If content is falsy, throw `new Error('Empty response from AI')`
- Return the content string
- Do NOT catch errors — let AbortError and FetchError propagate to the caller (ScribePopover handles them)

**4. `deriveLoadingMessage(actionId, label)` function:**
- For `actionId === 'free-prompt'`: return `'Processing...'` (per Pitfall 4: label is the user's full prompt, not suitable for display)
- For actions starting with `translate-`: return `'Translating to ${label}...'`
- For all others: derive from label by converting to gerund form:
  - `'Correct grammar'` -> `'Correcting grammar...'`
  - `'Make it shorter'` -> `'Making it shorter...'`
  - Simple heuristic: take first word, add "-ing" if it doesn't already end in "-ing", append rest + "..."
  - If the heuristic produces something odd, fall back to `'${label}...'`
  - Keep it simple — a basic switch on known action labels is also acceptable.

**Import from scribeActions.js:** `SCRIBE_ACTIONS`, `FREE_PROMPT_CONFIG`, `buildTranslateChildren`

**Do NOT import cozy-client or cozy-stack-client** — the `client` object is passed as a parameter.

**Export all four:** `export { callScribeAI, buildMessages, deriveLoadingMessage, SYSTEM_PROMPT }`
