Currently `cellsToMarkdownTable` builds a grid from row 0 to maxRow, creating empty rows for unselected row indices. Fix by re-indexing:

```javascript
export function cellsToMarkdownTable(cells) {
  if (cells.length === 0) return ''

  // Collect unique row and column indices
  const rowSet = new Set()
  const colSet = new Set()
  for (const c of cells) {
    rowSet.add(c.row)
    colSet.add(c.col)
  }
  const rows = Array.from(rowSet).sort((a, b) => a - b)
  const cols = Array.from(colSet).sort((a, b) => a - b)

  // Build lookup: original index -> sequential index
  const rowMap = {}
  rows.forEach((r, i) => { rowMap[r] = i })
  const colMap = {}
  cols.forEach((c, i) => { colMap[c] = i })

  // Build 2D grid with sequential indices
  const grid = Array.from({ length: rows.length }, () =>
    Array(cols.length).fill('')
  )
  for (const c of cells) {
    grid[rowMap[c.row]][colMap[c.col]] = c.text.trim().replace(/\n/g, ' ')
  }

  // Build GFM pipe table
  const lines = []
  for (let r = 0; r < rows.length; r++) {
    lines.push('| ' + grid[r].join(' | ') + ' |')
    if (r === 0) {
      lines.push('| ' + grid[r].map(() => '---').join(' | ') + ' |')
    }
  }
  return lines.join('\n')
}
```

This ensures that selecting rows 2-4 of a table produces a 3-row preview table, not a 5-row table with empty header rows.
