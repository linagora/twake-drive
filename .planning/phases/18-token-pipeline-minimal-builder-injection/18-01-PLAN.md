   - Takes the array returned by `marked.lexer(md)`
   - Returns an array of `{type: "paragraph", runs: [{text, bold, italic}]}` objects
   - Uses a recursive inner function `flattenInline(tokens, parentBold, parentItalic)` that:
     - For `type: "text"` nodes: pushes `{text, bold: !!parentBold, italic: !!parentItalic}`
     - For `type: "strong"` nodes: recurses into `tok.tokens` with `parentBold = true`
     - For `type: "em"` nodes: recurses into `tok.tokens` with `parentItalic = true`
     - For other nodes with `.tokens`: recurses, passing through current bold/italic state
     - For other leaf nodes with `.text`: treats as plain text with inherited formatting
   - At the top level, iterates `markedTokens`:
     - `type: "paragraph"` -> flattens inline tokens into runs
     - `type: "space"` -> skips (implicit paragraph separator)
     - Any other block type (heading, list, etc.) -> treat as paragraph fallback (flatten its tokens if present, or use raw text). This ensures graceful degradation in Phase 18 scope.

Use ES5 syntax throughout since this is inside code.js (even though plugin iframe supports modern JS -- keep consistency with the rest of the file). Use `var` not `const`/`let`, no arrow functions.

The function signature: `function flattenTokens(markedTokens) { ... }` -- a named function declaration inside the IIFE scope. See the research document's code example for the exact implementation pattern.

var md = '**bold** and *italic* and ***both***\n\nSecond paragraph with **strong**.';

// Inline the flattenTokens logic for verification
function flattenTokens(markedTokens) {
  var blocks = [];
  function flattenInline(tokens, parentBold, parentItalic) {
    var runs = [];
    for (var i = 0; i < tokens.length; i++) {
      var tok = tokens[i];
      if (tok.type === 'text') {
        runs.push({ text: tok.text, bold: !!parentBold, italic: !!parentItalic });
      } else if (tok.type === 'strong') {
        runs = runs.concat(flattenInline(tok.tokens, true, parentItalic));
      } else if (tok.type === 'em') {
        runs = runs.concat(flattenInline(tok.tokens, parentBold, true));
      } else if (tok.tokens) {
        runs = runs.concat(flattenInline(tok.tokens, parentBold, parentItalic));
      } else if (tok.text) {
        runs.push({ text: tok.text, bold: !!parentBold, italic: !!parentItalic });
      }
    }
    return runs;
  }
  for (var i = 0; i < markedTokens.length; i++) {
    var block = markedTokens[i];
    if (block.type === 'paragraph') {
      blocks.push({ type: 'paragraph', runs: flattenInline(block.tokens || [], false, false) });
    } else if (block.type === 'space') {
      // skip
    } else if (block.tokens) {
      blocks.push({ type: 'paragraph', runs: flattenInline(block.tokens || [], false, false) });
    } else if (block.text) {
      blocks.push({ type: 'paragraph', runs: [{ text: block.text, bold: false, italic: false }] });
    }
  }
  return blocks;
}
