```javascript
export function parseTableBlocks(text) {
  const regex = /\[TABLE:(\d+)\]([\s\S]*?)\[\/TABLE\]/g
  const tables = []
  let match
  while ((match = regex.exec(text)) !== null) {
    const index = parseInt(match[1], 10)
    const cells = parseCellMarkers(match[2])
    tables.push({ index, cells })
  }
  return tables
}
```
