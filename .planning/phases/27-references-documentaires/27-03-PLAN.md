```javascript
var crossRefMeta = JSON.parse(Asc.scope.crossRefMeta || "{}");

function recreateCrossRef(para, name) {
  var meta = crossRefMeta[name];
  if (!meta) return false;
  if (meta.type === "heading") {
    var headingParas = doc.GetAllHeadingParagraphs();
    // Find target heading by text match
    for (var hp = 0; hp < headingParas.length; hp++) {
      if (headingParas[hp].GetText() === meta.targetText) {
        return para.AddHeadingCrossRef(meta.refType || "text", headingParas[hp], true, false);
      }
    }
  } else if (meta.type === "bookmark") {
    return para.AddBookmarkCrossRef(meta.refType || "text", meta.targetName, true, false);
  } else if (meta.type === "footnote") {
    var fnParas = doc.GetFootnotesFirstParagraphs();
    if (meta.targetIndex < fnParas.length) {
      return para.AddFootnoteCrossRef(meta.refType || "formFootnoteNum", fnParas[meta.targetIndex], true, false);
    }
  }
  return false;
}
```
