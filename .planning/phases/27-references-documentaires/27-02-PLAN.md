if (tok.text && tok.text.indexOf("[^scribe-fn-") !== -1) {
  var fnRegex = /\[\^(scribe-fn-\d+)\]/g;
  var fnMatch;
  var fnLastIdx = 0;
  var fnText = tok.text;
  while ((fnMatch = fnRegex.exec(fnText)) !== null) {
    // Text before the marker
    if (fnMatch.index > fnLastIdx) {
      runs.push({ text: fnText.substring(fnLastIdx, fnMatch.index), bold: !!parentBold, italic: !!parentItalic, strikethrough: !!parentStrikethrough, underline: !!parentUnderline, code: !!parentCode, link: parentLink || null });
    }
    // The footnote marker run
    runs.push({ text: "", footnoteMarker: fnMatch[1], bold: false, italic: false, strikethrough: false, underline: false, code: false, link: null });
    fnLastIdx = fnRegex.lastIndex;
  }
  // Text after last marker
  if (fnLastIdx < fnText.length) {
    runs.push({ text: fnText.substring(fnLastIdx), bold: !!parentBold, italic: !!parentItalic, strikethrough: !!parentStrikethrough, underline: !!parentUnderline, code: !!parentCode, link: parentLink || null });
