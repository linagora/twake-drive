```javascript
/**
 * Validate a selection position message.
 * @param {*} msg - Message to validate
 * @returns {boolean} True if valid
 */
export function validateSelectionPosition(msg) {
  if (!msg || typeof msg !== 'object') return false
  if (msg.type !== MSG_TYPE_SELECTION_POSITION) return false
  if (msg.version !== PROTOCOL_VERSION) return false
  if (typeof msg.source !== 'string' || !msg.source) return false
  if (!msg.data || typeof msg.data !== 'object') return false
  if (typeof msg.data.top !== 'number' || typeof msg.data.left !== 'number') return false
  return true
}
```

**ScribeFloatingButton.jsx (NEW):**

Create a pill-shaped floating button rendered via React portal that appears near the text selection above the OO editor iframe.

```jsx
import React, { useEffect, useState } from 'react'
import { createPortal } from 'react-dom'
import PropTypes from 'prop-types'

/**
 * ScribeFloatingButton - Pill-shaped floating button rendered via portal.
 * Appears near the text selection with a fade-in animation.
 * Per user decisions:
 * - Icon + "Scribe" label (sparkle icon for discoverability)
 * - Pill-shaped with subtle elevation/shadow
 * - Smooth fade-in transition (~150-200ms)
 *
 * @param {Object} props
 * @param {{ top: number, left: number }|null} props.position - Fixed viewport coordinates
 * @param {boolean} props.visible - Whether the button should be shown
 * @param {Function} props.onClick - Click handler to trigger Scribe popover
 */
const ScribeFloatingButton = ({ position, visible, onClick }) => {
  // Animate: track mounted-and-visible for CSS transition
  const [show, setShow] = useState(false)

  useEffect(() => {
    if (visible && position) {
      // Trigger fade-in on next frame for CSS transition
      const raf = requestAnimationFrame(() => setShow(true))
      return () => cancelAnimationFrame(raf)
    } else {
      setShow(false)
    }
  }, [visible, position])

  if (!visible || !position) return null

  // Position button above the mouse position (offset upward)
  const BUTTON_OFFSET_Y = 10 // pixels above the mouse position
  const buttonStyle = {
    position: 'fixed',
    top: position.top - BUTTON_OFFSET_Y,
    left: position.left,
    transform: 'translate(-50%, -100%)', // center horizontally, position above
    zIndex: 99999,
    pointerEvents: 'auto',
    opacity: show ? 1 : 0,
    transition: 'opacity 180ms ease-in',
  }

  return createPortal(
    <div style={buttonStyle}>
      <button className="scribe-floating-trigger" onClick={onClick} type="button">
        {/* Sparkle icon (inline SVG matching plugin icon) */}
        <svg
          width="16"
          height="16"
          viewBox="0 0 24 24"
          fill="none"
          stroke="currentColor"
          strokeWidth="2"
          strokeLinecap="round"
          strokeLinejoin="round"
        >
          <path d="M9.937 15.5A2 2 0 0 0 8.5 14.063l-6.135-1.582a.5.5 0 0 1 0-.962L8.5 9.936A2 2 0 0 0 9.937 8.5l1.582-6.135a.5.5 0 0 1 .963 0L14.063 8.5A2 2 0 0 0 15.5 9.937l6.135 1.582a.5.5 0 0 1 0 .963L15.5 14.063a2 2 0 0 0-1.437 1.437l-1.582 6.135a.5.5 0 0 1-.963 0z" />
          <path d="M20 3v4" />
          <path d="M22 5h-4" />
        </svg>
        <span>Scribe</span>
      </button>
    </div>,
    document.body
  )
}

ScribeFloatingButton.propTypes = {
  position: PropTypes.shape({
    top: PropTypes.number.isRequired,
    left: PropTypes.number.isRequired,
  }),
  visible: PropTypes.bool.isRequired,
  onClick: PropTypes.func.isRequired,
}

export { ScribeFloatingButton }
```
